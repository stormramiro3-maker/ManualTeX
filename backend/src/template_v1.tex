% 9. CAJAS TEMÁTICAS
% ════════════════════════════════════════════════════════════════
% Seis tipos exactos. Cada uno con identidad visual ÚNICA.
% Breakable: definicion/formula/importante/nota = NO,
%            ejemplo/derivacion = SÍ (pueden ser largas).
%
\usepackage[most]{tcolorbox}

\tcbset{
  enhanced,
  boxrule=0.8pt,
  arc=1.5mm,
  left=2mm,
  right=2mm,
  top=1.5mm,
  bottom=1.5mm,
  before skip=12pt,
  after skip=12pt,
  fonttitle=\bfseries,
  coltitle=white
}

% AZUL — Definición: conceptos, leyes, principios
\newtcolorbox{definicion}[1][]{
  colback=fondoDefinicion,
  colframe=bordeDefinicion,
  colbacktitle=bordeDefinicion,
  title={Definición},
  breakable=false,
  #1
}

% LILA — Fórmula: ecuaciones clave con nombre
\newtcolorbox{formula}[1][]{
  colback=fondoFormula,
  colframe=bordeFormula,
  colbacktitle=bordeFormula,
  title={Fórmula},
  breakable=false,
  #1
}

% ROJO — Importante: advertencias, errores frecuentes
\newtcolorbox{importante}[1][]{
  colback=fondoImportante,
  colframe=bordeImportante,
  colbacktitle=bordeImportante,
  title={Importante},
  breakable=false,
  #1
}

% AMARILLO — Nota: contexto adicional, aclaraciones
\newtcolorbox{nota}[1][]{

  colframe=bordeNota,
  colbacktitle=bordeNota,

  #1
}
